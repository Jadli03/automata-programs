<script>


  const app = angular.module("myApp",[]);
	app.controller ('stu',function($scope){
	 $scope.fanme= "";
	 $scope.age = 0;
         $scope.course = "null";
   	console.log($scope.age);
	 $scope.fDetail = function () {
		return ("Full Name:" + $scope.fname 
				+ " " + 
			"Age: " + $scope.age 
				+ " " + 
			"Course: " + $scope.course);		
	};

	$scope.validate = function () {
		if($scope.fname == undefined ||
	           $scope.age <= 0 || 
		   $scope.course == "null")
			return "";
		if($scope.age < 18 && $scope.course == "BCA")
			return "You are Not Eligible";
		else if($scope.age < 20 && $scope.course == "MCA") 
			return "You are Not Eligible";
		else 
			return "You are Eligible"; 
	}
   });
